Beispielhafte Enwicklung Meta Heuristiken: ACO

  1. Calculate distance between all cities (l. 2)
  2. Find the x nearest cities (l. 3)
  3. Let each ants calculate one route (l. 6)
  4. From all routes save the shortest (l. 7)
  5. Update the pheromone accordingly (l. 9+10)

1 main {
2    mkt::roi_start();
3    distance.mapIndexInPlace(calculate_distance());
4    city.mapIndex(calculate_iroulette());
5    for (int i = 0; i < iterations; ++i) {
6       antss.mapIndex(route_kernel());
7       bestroute = d_routes_distance.reduce(min);
8       antss.mapIndex(update_best_sequence(bestroute));
9       antss.mapIndex(update_delta_phero());
10      city.mapIndex(update_phero());
11    }
12    mkt::print(distance);
13    mkt::roi_end();
14 }
